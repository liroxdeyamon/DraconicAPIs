    NAME: "Animates"
    AFFECTED -> [MAP.E, MAP.R, MAP.MON, MAP.I]
    FLAT -> {
        NAME -> String[]

    }
}

    NAME -> String[]
    SHORT -> String[]
}

NOUNS -> {
    MAP -> {"name": Noun(), "name": {1: Noun(), 2: Noun(), ...}, ...},
    FLAT -> Noun[],
    SUFFIXES -> {
        MAP: {
            [MOODS.D]: {
                [GENDERS.E.NAME]: {
                    [NUMBERS.S]: { 1: "", 2: "", 3: "", 4: "" },
                    [NUMBERS.D]: { ... },
                    [NUMBERS.P]: { ... }
                },
                [GENDERS.R.NAME]: { ... },
                [GENDERS.MON.NAME]: { ... },
                [GENDERS.I.NAME]: { ... },
                [GENDERS.MAG.NAME]: { ... },
                [GENDERS.MUN.NAME]: { ... },
                [GENDERS.A.NAME]: { ... }
            },
            [MOODS.R]: { ... }
        }
        FLAT -> String[],
    },
    fetch(keyword) -> Noun[] // all nouns with keyword in it
    fetchByDefinition(def) -> Noun[] // all nouns with keyword in its definition
}
